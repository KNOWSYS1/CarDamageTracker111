<body>
    <h1>Car Damage Tracker</h1>
    <div class="main-content">
        <div class="container">
            <img id="carImage" src="https://static.wixstatic.com/media/979173_907aaba065dc4728bf11df75c5581a3d~mv2.jpg/v1/fill/w_395,h_600,al_c,q_80,enc_avif,quality_auto/979173_907aaba065dc4728bf11df75c5581a3d~mv2.jpg" alt="Car Diagram" width="395" height="600">
        </div>
        <div id="damageList">
            <h3>Damage List</h3>
            <ul id="damageItems"></ul>
        </div>
    </div>
    <br>
    <button onclick="saveDamages()">Save Damages</button>
    <button onclick="loadDamages()">Load Previous Damages</button>

    <!-- Modal for entering damage details -->
    <div id="damageModal" style="display: none;">
        <h3>Enter Damage Details</h3>
        <label>Date: <input type="date" id="damageDate"></label><br>
        <label>Time: <input type="time" id="damageTime"></label><br>
        <label>Description: <textarea id="damageDescription" rows="4" cols="30"></textarea></label><br>
        <button onclick="submitDamageDetails()">Submit</button>
        <button onclick="closeModal()">Cancel</button>
    </div>
    <div id="modalBackdrop" style="display: none;"></div>

    <script>
        let damages = [];
        let currentClick = null;
        const passkey = "delete123"; // Generic passkey for deleting damages

        function init() {
            console.log('Initializing car damage tracker');
            const img = document.getElementById('carImage');
            if (!img) {
                console.error('Car image element not found');
                return;
            }
            img.onclick = getClickPosition;
            loadDamages();
        }

        function getClickPosition(e) {
            const pos = getCoordinates(e);
            if (pos) {
                console.log('Click position:', pos);
                currentClick = pos;
                showModal();
            }
        }

        function getCoordinates(e) {
            let PosX = 0, PosY = 0;
            const img = document.getElementById('carImage');
            const ImgPos = findPosition(img);
            if (!e) e = window.event;
            if (e.pageX || e.pageY) {
                PosX = e.pageX;
                PosY = e.pageY;
            } else if (e.clientX || e.clientY) {
                PosX = e.clientX + document.body.scrollLeft + document.documentElement.scrollLeft;
                PosY = e.clientY + document.body.scrollTop + document.documentElement.scrollTop;
            }
            PosX -= ImgPos[0];
            PosY -= ImgPos[1];
            return {x: PosX, y: PosY};
        }

        function findPosition(oElement) {
            let posX = 0, posY = 0;
            for (; oElement; oElement = oElement.offsetParent) {
                posX += oElement.offsetLeft;
                posY += oElement.offsetTop;
            }
            return [posX, posY];
        }

        function addMarker(x, y, id) {
            const marker = document.createElement('div');
            marker.className = 'marker';
            marker.id = `marker-${id}`;
            marker.style.left = `${x - 5}px`;
            marker.style.top = `${y - 5}px`;
            document.querySelector('.container').appendChild(marker);
            console.log('Marker added at:', x, y, 'ID:', id);
        }

        function showModal() {
            const modal = document.getElementById('damageModal');
            const backdrop = document.getElementById('modalBackdrop');
            if (modal && backdrop) {
                modal.style.display = 'block';
                backdrop.style.display = 'block';
                document.getElementById('damageDate').value = '';
                document.getElementById('damageTime').value = '';
                document.getElementById('damageDescription').value = '';
                console.log('Modal opened');
            } else {
                console.error('Modal or backdrop element not found');
            }
        }

        function closeModal() {
            const modal = document.getElementById('damageModal');
            const backdrop = document.getElementById('modalBackdrop');
            if (modal && backdrop) {
                modal.style.display = 'none';
                backdrop.style.display = 'none';
                currentClick = null;
                console.log('Modal closed');
            }
        }

        function submitDamageDetails() {
            const date = document.getElementById('damageDate').value;
            const time = document.getElementById('damageTime').value;
            const description = document.getElementById('damageDescription').value;
            if (currentClick && date && time && description) {
                const id = damages.length; // Use index as unique ID
                const damage = {
                    id: id,
                    x: currentClick.x,
                    y: currentClick.y,
                    date: date,
                    time: time,
                    description: description
                };
                damages.push(damage);
                addMarker(currentClick.x, currentClick.y, id);
                updateDamageList();
                console.log('Damage added:', damage);
                closeModal();
            } else {
                alert('Please fill in all fields.');
                console.error('Missing fields:', { date, time, description });
            }
        }

        function deleteDamage(index) {
            const enteredPasskey = prompt('Enter passkey to delete this damage:');
            if (enteredPasskey === passkey) {
                const damage = damages[index];
                if (damage) {
                    damages.splice(index, 1);
                    // Reassign IDs to remaining damages
                    damages.forEach((d, i) => d.id = i);
                    // Remove all markers and re-render
                    document.querySelectorAll('.marker').forEach(m => m.remove());
                    damages.forEach(d => addMarker(d.x, d.y, d.id));
                    updateDamageList();
                    saveDamages();
                    console.log('Damage deleted at index:', index);
                }
            } else {
                alert('Incorrect passkey. Deletion cancelled.');
                console.log('Incorrect passkey entered for deletion');
            }
        }

        function updateDamageList() {
            const list = document.getElementById('damageItems');
            if (!list) {
                console.error('Damage list element not found');
                return;
            }
            list.innerHTML = '';
            damages.forEach((d, index) => {
                const li = document.createElement('li');
                li.innerHTML = `
                    <span><strong>Damage ${index + 1}</strong>: (${d.x}, ${d.y})<br>Date: ${d.date}<br>Time: ${d.time}<br>Description: ${d.description}</span>
                    <button onclick="deleteDamage(${index})">Delete</button>
                `;
                list.appendChild(li);
            });
            console.log('Damage list updated:', damages);
        }

        function saveDamages() {
            try {
                localStorage.setItem('carDamages', JSON.stringify(damages));
                alert('Damages saved to local storage!');
                console.log('Damages saved to localStorage:', damages);
            } catch (e) {
                console.error('Error saving to localStorage:', e);
                alert('Error saving damages. Check browser settings.');
            }
        }

        function loadDamages() {
            try {
                const saved = localStorage.getItem('carDamages');
                if (saved) {
                    damages = JSON.parse(saved);
                    // Ensure IDs are set
                    damages.forEach((d, i) => d.id = i);
                    damages.forEach(d => addMarker(d.x, d.y, d.id));
                    updateDamageList();
                    console.log('Damages loaded from localStorage:', damages);
                } else {
                    console.log('No saved damages found');
                }
            } catch (e) {
                console.error('Error loading from localStorage:', e);
                alert('Error loading damages. Check browser settings.');
            }
        }

        init();
    </script>
